235 void _rgb2label(const cv::Mat &src, unsigned char *dst, const unsigned char (*rgb2labelMap)[3],
236  unsigned int nClasses)
237 {
238  for (unsigned int v=0; v<src.rows; v++)
239  {
240  for (unsigned int u=0; u<src.cols; u++)
241  {
242  unsigned char l=0;
243  if (src.channels()==3)
244  {
245  cv::Vec3b labelValue = src.at<cv::Vec3b>(v, u);
246  for (; l<nClasses; l++)
247  {
248  if (labelValue[0]==rgb2labelMap[l][0] &&
249  labelValue[1]==rgb2labelMap[l][1] &&
250  labelValue[2]==rgb2labelMap[l][2]) break;
251  }
252  dst[v*src.cols+u] = (l<nClasses) ? l+1 : 0;
253  }
254  else
255  {
256  cv::Vec4b labelValue = src.at<cv::Vec4b>(v, u);
257  for (; l<nClasses; l++)
258  {
259  if (labelValue[0]==rgb2labelMap[l][0] &&
260  labelValue[1]==rgb2labelMap[l][1] &&
261  labelValue[2]==rgb2labelMap[l][2]) break;
262  }
263  dst[v*src.cols+u] = (l<nClasses) ? l+1 : 0;
264  }
265  }
266  }
267 }

271 cv::Rect _extractROI(const cv::Mat &img)
272 {
273  std::vector<std::vector<cv::Point> > contours;
274  std::vector<cv::Vec4i> hierarchy;
275  cv::Rect boundRect;
276 
277  cv::Mat mask = img>0;
278  cv::findContours(mask, contours, hierarchy, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0));
279 
280  // Return the bounding rect associated with the counter with the bigger area
281  boundRect = cv::boundingRect(contours[0]);
282  for (int i=1; i<contours.size(); i++)
283  {
284  cv::Rect currBoundRect = cv::boundingRect(contours[i]);
285  if (currBoundRect.width*currBoundRect.height > boundRect.width*boundRect.height)
286  {
287  boundRect = currBoundRect;
288  }
289  }
290 
291  return boundRect;
292 }
