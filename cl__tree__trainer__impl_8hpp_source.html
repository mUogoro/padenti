195 template <typename ImgType, unsigned int nChannels, typename FeatType, unsigned int FeatDim,
196  unsigned int nClasses>
197 void CLTreeTrainer<ImgType, nChannels, FeatType, FeatDim, nClasses>::train(
198  Tree<FeatType, FeatDim, nClasses> &tree,
199  const TrainingSet<ImgType, nChannels> &trainingSet,
200  const TreeTrainerParameters<FeatType, FeatDim> &params,
201  unsigned int startDepth, unsigned int endDepth)
202 {
204  if (startDepth!=1) throw "Starting depth must be equal to 1";
205 
206  _initTrain(tree, trainingSet, params, startDepth, endDepth);
207 
208 
209  for (unsigned int currDepth=startDepth; currDepth<endDepth; currDepth++)
210  {
211  boost::chrono::steady_clock::time_point perLevelTrainStart =
212  boost::chrono::steady_clock::now();
213 
214  unsigned int frontierSize = _initFrontier(tree, params, currDepth);
215  unsigned int nSlices = _initHistogram(params);
216 
217  if (nSlices>1)
218  {
219  BOOST_LOG_TRIVIAL(info) << "Maximum allowed global histogram size reached: split in "
220  << nSlices << " slices";
221  }
222 
223  // Flag all images as to-be-skipped: the flag will be set to false if at least one
224  // image pixel is processed
225  std::fill_n(m_toSkipTsImg, trainingSet.getImages().size(), true);
226 
227  for (unsigned int i=0; i<nSlices; i++)
228  {
229  _traverseTrainingSet(trainingSet, params, currDepth, i);
230  _learnBestFeatThr(tree, params, currDepth, i);
231  }
232 
233  // Update skipped images flags
234  std::copy(m_toSkipTsImg, m_toSkipTsImg+trainingSet.getImages().size(), m_skippedTsImg);
235 
236 
237  boost::chrono::duration<double> perLevelTrainTime =
238  boost::chrono::duration_cast<boost::chrono::duration<double> >(boost::chrono::steady_clock::now() -
239  perLevelTrainStart);
240  BOOST_LOG_TRIVIAL(info) << "Depth " << currDepth << " trained in "
241  << perLevelTrainTime.count() << " seconds";
242  }
243 }
