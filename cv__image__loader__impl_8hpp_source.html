243 void _rgb2label(const cv::Mat &src, unsigned char *dst, const unsigned char (*rgb2labelMap)[3],
244  unsigned int nClasses)
245 {
246  for (unsigned int v=0; v<src.rows; v++)
247  {
248  for (unsigned int u=0; u<src.cols; u++)
249  {
250  unsigned char l=0;
251  if (src.channels()==3)
252  {
253  cv::Vec3b labelValue = src.at<cv::Vec3b>(v, u);
254  for (; l<nClasses; l++)
255  {
256  if (labelValue[0]==rgb2labelMap[l][0] &&
257  labelValue[1]==rgb2labelMap[l][1] &&
258  labelValue[2]==rgb2labelMap[l][2]) break;
259  }
260  dst[v*src.cols+u] = (l<nClasses) ? l+1 : 0;
261  }
262  else
263  {
264  cv::Vec4b labelValue = src.at<cv::Vec4b>(v, u);
265  for (; l<nClasses; l++)
266  {
267  if (labelValue[0]==rgb2labelMap[l][0] &&
268  labelValue[1]==rgb2labelMap[l][1] &&
269  labelValue[2]==rgb2labelMap[l][2]) break;
270  }
271  dst[v*src.cols+u] = (l<nClasses) ? l+1 : 0;
272  }
273  }
274  }
275 }
