39  boost::chrono::steady_clock::now();

41  size_t perNodeHistogramSize = nClasses*params.nFeatures*params.nThresholds;
42  unsigned int frontierSize = m_frontierIdxMap.size();
43  unsigned int toTrainNodes = ((currSlice+1)*m_histogramSize > frontierSize) ?
44  (frontierSize%m_histogramSize) : m_histogramSize;
45 
46  size_t learnBuffSize =
47  ((PARALLEL_LEARNT_NODES<frontierSize) ? PARALLEL_LEARNT_NODES : frontierSize) *
48  (params.nFeatures*params.nThresholds)/PER_THREAD_FEAT_THR_PAIRS;
49  if (!currSlice)
50  {
51  std::fill_n(m_bestFeatures, learnBuffSize, 0);
52  std::fill_n(m_bestThresholds, learnBuffSize, 0);
53  std::fill_n(m_bestEntropies, learnBuffSize, -1.);
54  }

56 
57  unsigned int nIters = toTrainNodes/PARALLEL_LEARNT_NODES + \
58  ((toTrainNodes%PARALLEL_LEARNT_NODES) ? 1 : 0);
59  for (unsigned int i=0; i<nIters; i++)
60  {
61  unsigned int currNNodes = (!i && toTrainNodes<=PARALLEL_LEARNT_NODES) ? toTrainNodes :
62  ((i==(nIters-1) && toTrainNodes%PARALLEL_LEARNT_NODES) ?
63  (toTrainNodes%PARALLEL_LEARNT_NODES) : PARALLEL_LEARNT_NODES);
64  unsigned int nThreads =
65  currNNodes*(params.nFeatures*params.nThresholds)/PER_THREAD_FEAT_THR_PAIRS;
66  unsigned int frontierOffset = currSlice*m_histogramSize + i*PARALLEL_LEARNT_NODES;
67 
68  for (unsigned int n=0; n<currNNodes; n++)
69  {
70  unsigned int nodeID = m_frontier[frontierOffset+n];
71  m_clQueue1.enqueueWriteBuffer(m_clHistogramBuff,
72  CL_FALSE,
73  n*perNodeHistogramSize*sizeof(cl_uint),
74  perNodeHistogramSize*sizeof(cl_uint),
75  (void*)m_histogram[i*PARALLEL_LEARNT_NODES+n]);
76 
77  // Upload per-node per-class total number of samples on GPU
78  m_clQueue1.enqueueWriteBuffer(m_clPerClassTotSamplesBuff,
79  CL_FALSE,
80  n*nClasses*sizeof(cl_uint),
81  nClasses*sizeof(cl_uint),
82  (void*)(&m_perClassTotSamples[nodeID*nClasses]));
83  }
84 
86  m_clQueue1.enqueueNDRangeKernel(m_clLearnBestFeatKern,
87  cl::NullRange,
88  cl::NDRange(nThreads),
89  cl::NDRange(32));
90 
91  m_clQueue1.enqueueReadBuffer(m_clBestFeaturesBuff,
92  CL_FALSE,
93  0, nThreads*sizeof(cl_uint), m_bestFeatures);
94  m_clQueue1.enqueueReadBuffer(m_clBestThresholdsBuff,
95  CL_FALSE,
96  0, nThreads*sizeof(cl_uint), m_bestThresholds);
97  m_clQueue1.enqueueReadBuffer(m_clBestEntropiesBuff,
98  CL_TRUE,
99  0, nThreads*sizeof(cl_float), m_bestEntropies);
100 
101  unsigned int perNodeThreads = nThreads/currNNodes;
102  for (unsigned int n=0; n<currNNodes; n++)
103  {
104  unsigned int *perNodeBestFeatures = &m_bestFeatures[perNodeThreads*n];
105  unsigned int *perNodeBestThresholds = &m_bestThresholds[perNodeThreads*n];
106  float *perNodeBestEntropies = &m_bestEntropies[perNodeThreads*n];
107 
108  unsigned int nodeID = m_frontier[frontierOffset+n];
109  unsigned int perNodeSliceOffset = i*PARALLEL_LEARNT_NODES+n;
110 
111  unsigned int bestID = std::distance(perNodeBestEntropies,
112  std::max_element(perNodeBestEntropies, perNodeBestEntropies+perNodeThreads));
113 
114  // Update children histograms and left children
117  const TreeNode<FeatType, FeatDim> &currNode = tree.getNode(nodeID);
118  const TreeNode<FeatType, FeatDim> &leftChildNode = tree.getNode(nodeID*2+1);
119  const TreeNode<FeatType, FeatDim> &rightChildNode = tree.getNode(nodeID*2+2);
120  unsigned int *leftHistogram = new unsigned int[nClasses];
121  unsigned int *rightHistogram = new unsigned int[nClasses];
122  unsigned int lSum=0, rSum=0;
123  for (unsigned int l=0; l<nClasses; l++)
124  {
125  unsigned int *currHistogram = m_histogram[perNodeSliceOffset];
126  /*
127  unsigned int offset =
128  l * (params.nFeatures*params.nThresholds) +
129  perNodeBestFeatures[bestID] * (params.nThresholds) +
130  perNodeBestThresholds[bestID];
131  */
132  unsigned int offset =
133  l * (params.nThresholds*params.nFeatures) +
134  perNodeBestThresholds[bestID] * (params.nFeatures) +
135  perNodeBestFeatures[bestID];
136  leftHistogram[l] = currHistogram[offset];
137  rightHistogram[l] = m_perClassTotSamples[nodeID*nClasses+l]-leftHistogram[l];
138  lSum += leftHistogram[l];
139  rSum += rightHistogram[l];
140  }
141 
142  // If all the pixels follow the same path, the best-feature is not discriminative enough
143  // and the current node is kept as a leaf node
145  if (!lSum || !rSum) continue;
146 
147  // Update the total number of samples reaching child nodes
148  m_perNodeTotSamples[nodeID*2+1] = lSum;
149  m_perNodeTotSamples[nodeID*2+2] = rSum;
150  for (unsigned int l=0; l<nClasses; l++)
151  {
152  m_perClassTotSamples[(nodeID*2+1)*nClasses+l]=leftHistogram[l];
153  m_perClassTotSamples[(nodeID*2+2)*nClasses+l]=rightHistogram[l];
154  assert((leftHistogram[l]+rightHistogram[l])==m_perClassTotSamples[nodeID*nClasses+l]);
155  }
156  assert((lSum+rSum)==m_perNodeTotSamples[nodeID]);
157 
158  float *leftPosteriors = new float[nClasses];
159  float *rightPosteriors = new float[nClasses];
160  for (unsigned int l=0; l<nClasses; l++)
161  {
162  leftPosteriors[l] = ((float)leftHistogram[l])/lSum;
163  rightPosteriors[l] = ((float)rightHistogram[l])/rSum;
164  }
165 
166  std::copy(leftPosteriors, leftPosteriors+nClasses, leftChildNode.m_posterior);
167  std::copy(rightPosteriors, rightPosteriors+nClasses, rightChildNode.m_posterior);
168  *leftChildNode.m_leftChild=-1;
169  *rightChildNode.m_leftChild=-1;
170 
171 
172  // Finally, recompute feature and threshold values for current node
177  /*
178  unsigned int seed[4] = {tree.getID()^m_seed,
179  nodeID^m_seed,
180  perNodeBestFeatures[bestID]^m_seed,
181  m_seed};
182  */
183  unsigned int seed[4] = {tree.getID(),
184  nodeID,
185  perNodeBestFeatures[bestID],
186  0};
187  unsigned int state[4];
188  for (unsigned int j=0; j<FeatDim; j+=4)
189  {
190  md5Rand(seed, state);
191 
192  currNode.m_feature[j] = params.featLowBounds[j] +
193  (FeatType)(((float)state[0])/(0xFFFFFFFF)*(params.featUpBounds[j]-params.featLowBounds[j]));
194  if ((j+1)>=FeatDim) break;
195 
196  currNode.m_feature[j+1] = params.featLowBounds[j+1] +
197  (FeatType)(((float)state[1])/(0xFFFFFFFF)*(params.featUpBounds[j+1]-params.featLowBounds[j+1]));
198  if ((j+2)>=FeatDim) break;
199 
200  currNode.m_feature[j+2] = params.featLowBounds[j+2] +
201  (FeatType)(((float)state[2])/(0xFFFFFFFF)*(params.featUpBounds[j+2]-params.featLowBounds[j+2]));
202  if ((j+3)>=FeatDim) break;
203 
204  currNode.m_feature[j+3] = params.featLowBounds[j+3] +
205  (FeatType)(((float)state[3])/(0xFFFFFFFF)*(params.featUpBounds[j+3]-params.featLowBounds[j+3]));
206 
207  std::copy(state, state+4, seed);
208  }
209 
210 
211  seed[0] = tree.getID();
212  seed[1] = nodeID;
213  seed[2] = perNodeBestFeatures[bestID];
214  seed[3] = 1;
215  for (unsigned int j=0; j<(perNodeBestThresholds[bestID]/4+1); j++)
216  {
217  md5Rand(seed, state);
218  std::copy(state, state+4, seed);
219  }
220  *currNode.m_threshold = params.thrLowBound +
221  (FeatType)((float)state[perNodeBestThresholds[bestID]%4]/0xFFFFFFFF*
222  (params.thrUpBound-params.thrLowBound));
223 
224 
225  /*
226  *currNode.m_threshold = params.thrLowBound +
227  (FeatType)((float)perNodeBestThresholds[bestID]*
228  (params.thrUpBound-params.thrLowBound)/params.nThresholds);
229  */
230 
231 
232  // Update current node left child
233  *currNode.m_leftChild = nodeID*2+1;
234 
235 
236  // Done with node
237 
238  delete []rightPosteriors;
239  delete []leftPosteriors;
240  delete []rightHistogram;
241  delete []leftHistogram;
242  }
243  }
