108  // Generic feature type trick:
109  // define the OpenCL feature type using a typedef depending on the template-specified
110  // C feature type. Use type-traits to retrieve the typedef code and save it on a temporary
111  // file included by other OpenCL source.
112  // Note: only standard types are supported
113 #ifdef WIN32
114  char *tmpPath = getenv("Temp");
115  std::stringstream clFeatTypeFilePathSS;
116  clFeatTypeFilePathSS << tmpPath << "\\feat_type.cl";
117  std::ofstream clFeatTypeFile(clFeatTypeFilePathSS.str().c_str());
118 #else
119  std::ofstream clFeatTypeFile("/tmp/feat_type.cl");
120 #endif // WIN32
121  std::string featTypedefCode;
122  FeatTypeTrait<FeatType>::getCLTypedefCode(featTypedefCode);
123  clFeatTypeFile.write(featTypedefCode.c_str(), featTypedefCode.length());
124  clFeatTypeFile.close();
125 
126  // Do the same for the image type:
127  // - if the image contains up to 4 channel, work with image2d_t;
128  // - if the image has more than 4 channel, work with 3D images (i.e. image3d_t)
129 #ifdef WIN32
130  std::stringstream clImgTypeFilePathSS;

139  "typedef image3d_t image_t;\n");
140  std::string imgTypedefCodeFooter("\n#endif //__IMG_TYPE");
141  clImgTypeFile.write(imgTypedefCodeHeader.c_str(), imgTypedefCodeHeader.length());
142  clImgTypeFile.write(imgTypedefCode.c_str(), imgTypedefCode.length());
143  clImgTypeFile.write(imgTypedefCodeFooter.c_str(), imgTypedefCodeFooter.length());

147  std::stringstream opts;
148  //opts << "-I/tmp -Ikernels -I" << featureKernelPath;
149 #ifdef WIN32
150  opts << "-I" << tmpPath << " -I" << featureKernelPath;
151 #else

153 #endif // WIN32
154

166  }

171  }
172  catch (cl::Error e)
173  {
174  std::string buildLog;
175  m_clPredictProg.getBuildInfo(m_clDevice, CL_PROGRAM_BUILD_LOG, &buildLog);
176 
177  std::cerr << buildLog << std::endl;
178  throw buildLog;
179  }
180 
181  try
182  {
183  m_clLearnBestFeatProg.build(opts.str().c_str());
184  }
185  catch (cl::Error e)
186  {
187  std::string buildLog;
188  m_clLearnBestFeatProg.getBuildInfo(m_clDevice, CL_PROGRAM_BUILD_LOG, &buildLog);
189 
190  std::cerr << buildLog << std::endl;
191  throw buildLog;
192  }
193 
195  m_clPerImgHistKern = cl::Kernel(m_clHistUpdateProg, "computePerImageHistogram");
196  m_clPredictKern = cl::Kernel(m_clPredictProg, "predict");
197  m_clLearnBestFeatKern = cl::Kernel(m_clLearnBestFeatProg, "learnBestFeature");
198 }

201 template <typename ImgType, unsigned int nChannels, typename FeatType, unsigned int FeatDim,
202  unsigned int nClasses>
203 void CLTreeTrainer<ImgType, nChannels, FeatType, FeatDim, nClasses>::train(
204  Tree<FeatType, FeatDim, nClasses> &tree,
205  const TrainingSet<ImgType, nChannels> &trainingSet,
206  const TreeTrainerParameters<FeatType, FeatDim> &params,
207  unsigned int startDepth, unsigned int endDepth)
208 {
210  if (startDepth!=1) throw "Starting depth must be equal to 1";
211 
212  _initTrain(tree, trainingSet, params, startDepth, endDepth);
213 
214 
215  for (unsigned int currDepth=startDepth; currDepth<endDepth; currDepth++)
216  {
217  boost::chrono::steady_clock::time_point perLevelTrainStart =
218  boost::chrono::steady_clock::now();
219 
220  unsigned int frontierSize = _initFrontier(tree, params, currDepth);
221  unsigned int nSlices = _initHistogram(params);
222 
223 
224  if (nSlices>1)
225  {
226  BOOST_LOG_TRIVIAL(info) << "Maximum allowed global histogram size reached: split in "
227  << nSlices << " slices";
228  }
229 
230 
231  // Flag all images as to-be-skipped: the flag will be set to false if at least one
232  // image pixel is processed
233  std::fill_n(m_toSkipTsImg, trainingSet.getImages().size(), true);
234 
235  for (unsigned int i=0; i<nSlices; i++)
236  {
237  _traverseTrainingSet(trainingSet, params, currDepth, i);
238  _learnBestFeatThr(tree, params, currDepth, i);
239  }
240 
241  // Update skipped images flags
242  std::copy(m_toSkipTsImg, m_toSkipTsImg+trainingSet.getImages().size(), m_skippedTsImg);
243 
244 
245  boost::chrono::duration<double> perLevelTrainTime =
246  boost::chrono::duration_cast<boost::chrono::duration<double> >(boost::chrono::steady_clock::now() -
247  perLevelTrainStart);
248 
249  BOOST_LOG_TRIVIAL(info) << "Depth " << currDepth << " trained in "
250  << perLevelTrainTime.count() << " seconds";
251 
252  }
253 
254  _cleanTrain();
255 }
