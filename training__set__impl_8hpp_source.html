42 template <typename type, unsigned int nChannels>
43 TrainingSet<type, nChannels>::TrainingSet(const std::string &tsPathStr,
44  const std::string &dataSuffix, const std::string &labelsSuffix,
45  unsigned int nClasses,
46  ImageLoader<type, nChannels> &dataLoader,
47  ImageLoader<unsigned char, 1> &labelsLoader,
48  const ImageSampler<type, nChannels> &sampler):
49  m_nImages(0),
50  m_nClasses(nClasses)
51 {
52  std::vector<std::pair<std::string, std::string> > imgLabelsPairs;
53  path tsPath(tsPathStr);
54 
55  directory_iterator endItr;
56  for (directory_iterator itr(tsPath); itr!=endItr; ++itr)
57  {
58  if (!is_directory(itr->status()))
59  {
60  // Look if the current file is a training set sample
61  std::string fName = itr->path().filename().string();
62  unsigned int prefixLen = fName.size()-dataSuffix.size();
63  if (fName.rfind(dataSuffix)==prefixLen)
64  {
65  // Sample found!!! Look for the labels image
66  std::string labelsImgName = tsPathStr + "/" + fName.substr(0, prefixLen) + labelsSuffix;
67  if (exists(labelsImgName))
68  {
69  fName = tsPathStr + "/" + fName;
70  imgLabelsPairs.push_back(std::make_pair(fName, labelsImgName));
71  }
72  }
73  }
74  }
75 
80  type *imgBuff = new type[MAX_IMG_WIDTH*MAX_IMG_HEIGHT*MAX_IMG_CHANNELS];
81  unsigned int imgWidth, imgHeight;
82  unsigned char *labelsBuff = new unsigned char[MAX_IMG_WIDTH*MAX_IMG_HEIGHT];
83  unsigned int labelsWidth, labelsHeight;
84  unsigned int *samplesBuff = new unsigned int[MAX_IMG_WIDTH*MAX_IMG_HEIGHT];
85  unsigned int nSamples;
86 
87 
88  for (std::vector<std::pair<std::string, std::string> >::iterator it = imgLabelsPairs.begin();
89  it!=imgLabelsPairs.end(); ++it)
90  {
91  std::string imgFName = it->first;
92  std::string labelsFName = it->second;
93 
94  dataLoader.load(imgFName, imgBuff, &imgWidth, &imgHeight);
95  labelsLoader.load(labelsFName, labelsBuff, &labelsWidth, &labelsHeight);
96 
97  if (imgWidth!=labelsWidth || imgHeight!=labelsHeight)
98  {
99  BOOST_LOG_TRIVIAL(warning) << "Skip image pair " << imgFName << " - " << labelsFName
100  << " due to different size ("
101  << imgWidth << "X" << imgHeight << ", "
102  << labelsWidth << "X" << labelsHeight << ")";
103  continue;
104  }
105 
106  nSamples = sampler.sample(imgBuff, labelsBuff, imgWidth, imgHeight,
107  samplesBuff);
108 
109  m_images.push_back(TrainingSetImage<type, nChannels>(imgBuff, imgWidth, imgHeight,
110  labelsBuff, m_nClasses,
111  samplesBuff, nSamples));
112  }
113  m_nImages = m_images.size();
114 
116  m_priors = new float[m_nClasses];
117  std::fill_n(m_priors, m_nClasses, 0.0f);
118 
119  double *tmpPriors = new double[m_nClasses];
120  std::fill_n(tmpPriors, m_nClasses, 0.0f);
121  for (typename std::vector<TrainingSetImage<type, nChannels> >::const_iterator it=m_images.begin();
122  it!=m_images.end(); ++it)
123  {
124  const float *priors = it->getPriors();
125  for (unsigned int i=0; i<m_nClasses; i++)
126  {
127  tmpPriors[i] += priors[i];
128  }
129  }
130  for (int i=0; i<m_nClasses; i++)
131  {
132  m_priors[i] = static_cast<float>(tmpPriors[i]/m_nImages);
133  }
134 
135  delete []tmpPriors;
136  delete []samplesBuff;
137  delete []labelsBuff;
138  delete []imgBuff;
139 }

156 template <typename type, unsigned int nChannels>
157 TrainingSet<type, nChannels>& TrainingSet<type, nChannels>::operator<<(const TrainingSetImage<type, nChannels> &image)
158 {
159  m_images.push_back(image);
160 
161  // Update training set posterior probabilities
162  for (int c=0; c<m_nClasses; c++)
163  {
164  double pC = m_priors[c];
165  double pCImg = image.getPriors()[c];
166  m_priors[c] = static_cast<float>(pC*m_nImages/(m_nImages+1) +
167  pCImg/(m_nImages+1));
