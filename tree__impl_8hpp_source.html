101 template <typename FeatType, unsigned int FeatDim,
102  unsigned int nClasses>
103 void Tree<FeatType, FeatDim, nClasses>::load(const std::string &treePath,
104  int idx)
105 {
106  boost::property_tree::ptree pt;
107  boost::property_tree::read_xml(treePath, pt);
108 
109  if (idx!=-1)
110  {
111  m_id = idx;
112  }
113  unsigned int currDepth = pt.get<unsigned int>("Trees.MaxDepth");
114 
115  if (!m_depth || m_depth!=currDepth)
116  {
117  _clean();
118 
119  m_depth = currDepth;
120  _init();
121  }
122 
123  for (unsigned int i=0; i<((2<<(m_depth-1))-1); i++)
124  {
125  std::stringstream nodeStream;
126  const TreeNode<FeatType, FeatDim> &currNode = getNode(i);
127 
128  nodeStream << "Trees.Tree" << m_id << ".Node" << i << ".LeftChild";
129  *currNode.m_leftChild = pt.get<int>(nodeStream.str(), -2);
130 
131  if (*currNode.m_leftChild==-2)
132  {
133  // Node does not exists: fill with zeros
134  std::fill_n(currNode.m_feature, FeatDim, 0);
135  *currNode.m_threshold = 0;
136  std::fill_n(currNode.m_posterior, nClasses, 0);
137  }
138  else if (*currNode.m_leftChild==-1)
139  {
140  // Leaf node
141  std::stringstream histogramStream;
142 
143  nodeStream.str("");
144  nodeStream << "Trees.Tree" << m_id << ".Node" << i << ".Histogram";
145  histogramStream.str(pt.get<std::string>(nodeStream.str()));
146  for (int l=0; l<nClasses; l++)
147  {
148  histogramStream >> currNode.m_posterior[l];
149  }
150 
151  std::fill_n(currNode.m_feature, FeatDim, 0);
152  *currNode.m_threshold = 0;
153  }
154  else
155  {
156  // Split Node
157  std::stringstream histogramStream;
158  std::stringstream featuresStream;
159 
160  nodeStream.str("");
161  nodeStream << "Trees.Tree" << m_id << ".Node" << i << ".Histogram";
162  histogramStream.str(pt.get<std::string>(nodeStream.str()));
163  for (int l=0; l<nClasses; l++)
164  {
165  histogramStream >> currNode.m_posterior[l];
166  }
167 
168  nodeStream.str("");
169  nodeStream << "Trees.Tree" << m_id << ".Node" << i << ".SplitParameters";
170  featuresStream.str(pt.get<std::string>(nodeStream.str()));
171  for (int j=0; j<FeatDim; j++)
172  {
173  featuresStream >> currNode.m_feature[j];
174  }
175  featuresStream >> *currNode.m_threshold;
176  }
177  }
178 
179  // Done
180 }

186 template <typename FeatType, unsigned int FeatDim,
187  unsigned int nClasses>
188 void Tree<FeatType, FeatDim, nClasses>::save(const std::string &treePath,
189  int idx) const
190 {
191  boost::property_tree::ptree pt;
192  std::stringstream nodeStream;
193 
194  pt.put("Trees.TreeNumber", 1);
195  pt.put("Trees.MaxDepth", m_depth);
196 
197  for (unsigned int i=0; i<((2<<(m_depth-1))-1); i++)
198  {
199  const TreeNode<FeatType, FeatDim> &currNode = getNode(i);
200 
201  if (*currNode.m_leftChild==-2) continue;
202 
203  nodeStream.clear();
204  nodeStream.str("");
205  nodeStream << "Trees.Tree" << ((idx!=-1) ? idx : m_id) << ".Node" << i;
206  pt.add(nodeStream.str()+".LeftChild", *currNode.m_leftChild);
207 
208  if (*currNode.m_leftChild != -1)
209  {
210  // Split node
211  std::stringstream featThrStream;
212 
213  for (unsigned int f=0; f<FeatDim; f++)
214  {
215  featThrStream << currNode.m_feature[f] << " ";
216  }
217  featThrStream << (*currNode.m_threshold);
218 
219  pt.add(nodeStream.str()+".SplitParameters", featThrStream.str());
220  }
221 
222  std::stringstream histStream;
223 
224  for (unsigned int l=0; l<nClasses-1; l++)
225  {
226  histStream << currNode.m_posterior[l] << " ";
227  }
228  histStream << currNode.m_posterior[nClasses-1];
229 
230  pt.add(nodeStream.str()+".Histogram", histStream.str());
231  }
232 
233  boost::property_tree::write_xml(treePath, pt);
234 }

237 template <typename FeatType, unsigned int FeatDim,
238  unsigned int nClasses>
239 void Tree<FeatType, FeatDim, nClasses>::_init()
240 {
241  unsigned int nNodes = (2<<(m_depth-1))-1;
242 
243  // Allocate tree data. Tree nodes members are allocated separately on continuos
244  // vectors. This allow to easily pass tree nodes' data to GPU without dealing
245  // with host-GPU memory alignment problems when using array-of-structs to represent
246  // the tree
247  m_leftChildren = new int[nNodes];
248  m_features = new FeatType[nNodes*FeatDim];
249  m_thresholds = new FeatType[nNodes];
250  m_posteriors = new float[nNodes*nClasses];
251 
252  // To simplify node data access, provide an array-of-struct representation of the tree:
253  // structs members are pointers to the continuosly allocated data
254  m_nodes = new TreeNode<FeatType, FeatDim>[nNodes];
255  for (int i=0; i<nNodes; i++)
256  {
257  TreeNode<FeatType, FeatDim> *currNode = &m_nodes[i];
258 
259  currNode->m_leftChild = &m_leftChildren[i];
260  currNode->m_feature = &m_features[i*FeatDim];
261  currNode->m_threshold = &m_thresholds[i];
262  currNode->m_posterior = &m_posteriors[i*nClasses];
263 
264  // Init tree nodes:
265  // - -2 left child
266  // - zeroed feature, threshold and posteriors
267  *currNode->m_leftChild = -2;
268  std::fill_n(currNode->m_feature, FeatDim, (FeatType)0);
269  *currNode->m_threshold = (FeatType)0;
270  std::fill_n(currNode->m_posterior, nClasses, 0.0f);
271  }
272 }
