69  unsigned int startNode = m_frontier[frontierOffset];

71  (frontierSize%m_histogramSize) : m_histogramSize;
72  unsigned int endNode = m_frontier[frontierOffset+totNodes-1];

74  // Fill global histogram with zeros

76  for (int i=0; i<totNodes; i++) std::fill_n(m_histogram[i], perNodeHistogramSize, 0);

79  std::queue<int> fifoQueue;
80  pthread_mutex_t fifoMtx = PTHREAD_MUTEX_INITIALIZER;
81  pthread_cond_t fifoCond = PTHREAD_COND_INITIALIZER;
82  struct ConsumerProducerData<ImgType, nChannels, FeatType, FeatDim, nClasses> consumerProducerData;
83  consumerProducerData.nodesIDImg = m_clTsNodesIDImgPinnPtr;
84  consumerProducerData.maxImgWidth = m_maxTsImgWidth;
85  consumerProducerData.maxImgHeight = m_maxTsImgHeight;
86  consumerProducerData.perNodeTotSamples = m_perNodeTotSamples;
87  consumerProducerData.perClassTotSamples = m_perClassTotSamples;
88  consumerProducerData.frontierIdxMap = &m_frontierIdxMap;
89  consumerProducerData.histogram = m_histogram;
90  consumerProducerData.perImgHistogram = m_clPerImgHistBuffPinnPtr;
91  consumerProducerData.perImgHistogramStride = perImgHistogramStride;
92  consumerProducerData.trainingSet = &trainingSet;
93  consumerProducerData.skippedTsImg = m_skippedTsImg;
94  consumerProducerData.toSkipTsImg = m_toSkipTsImg;
95  consumerProducerData.params = &params;
96  consumerProducerData.currDepth = currDepth;
97  consumerProducerData.frontierOffset = frontierOffset;
98  consumerProducerData.startNode = startNode;
99  consumerProducerData.endNode = endNode;
100  consumerProducerData.fifoMtx = &fifoMtx;

105  // Start the consumer
106  int queueIdx=0;
107  pthread_t consumer;
108  pthread_create(&consumer, NULL,
109  _updateGlobalHistogram<ImgType, nChannels, FeatType, FeatDim, nClasses>,
110  &consumerProducerData);

114 
115  // OpenCL events for timing purposes
116  cl::Event startWriteEvent1, endWriteEvent1, startWriteEvent2, endWriteEvent2;
117  cl::Event startComputeEvent1, endComputeEvent1, startComputeEvent2, endComputeEvent2;
118  cl::Event startReadEvent, endReadEvent;
119  cl_ulong totWriteTime=0, totComputeTime=0, totReadTime=0;
120 
121  int imgID=0, fillWidth, fillHeight;
122  cl::size_t<3> origin, region;
123  origin[0]=0; origin[1]=0, origin[2]=0;
124 
125  // Iterate through images and update the histogram
126  const std::vector<TrainingSetImage<ImgType, nChannels> > &tsImages = trainingSet.getImages();
127  for (typename std::vector<TrainingSetImage<ImgType, nChannels> >::const_iterator it=tsImages.begin();
128  it!=tsImages.end(); ++it,++imgID)
129  {
130  const TrainingSetImage<ImgType, nChannels> &currImage = *it;
131 
132  cl::CommandQueue *weCLQueue = (imgID%2) ? &m_clQueue2 : &m_clQueue1;
133  cl::CommandQueue *rCLQueue = (imgID%2) ? &m_clQueue1 : &m_clQueue2;
134  size_t clPinnMemWOffset = (imgID%2) ? 1 : 0;
135 
136 
137  if (!m_skippedTsImg[imgID])
138  {
139  // ************ FIRST QUEUE: WRITE AND KERNELS LAUNCH *************/
140  fillWidth = (currImage.getWidth()%WG_PREDICT_WIDTH) ?
141  WG_PREDICT_WIDTH-(currImage.getWidth()%WG_PREDICT_WIDTH) : 0;
142  fillHeight = (currImage.getHeight()%WG_PREDICT_HEIGHT) ?
143  WG_PREDICT_HEIGHT-(currImage.getHeight()%WG_PREDICT_HEIGHT) : 0;
144  region[0]=m_maxTsImgWidth; region[1]=m_maxTsImgHeight; region[2]=1;
145 
146  // TODO: check if its faster to perform bound-check on coordinates and erase only
147  // the current image ROI instead of zeroing the whole image buffer
148  #ifdef CL_VERSION_1_2
149  cl_int4 zeroColor = {0, 0, 0, 0};
150  weCLQueue->enqueueFillImage((imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1,
151  zeroColor, origin, region,
152  NULL, (imgID%2) ? &startWriteEvent2 : &startWriteEvent1);
153  #else
154  // TODO: find a way to use pinned memory or zero-ing kernel
155  size_t rowPitch;
156  char *tmpNodesIDptr = (char*)weCLQueue->enqueueMapImage((imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1,
157  CL_TRUE, CL_MAP_WRITE,
158  origin, region, &rowPitch, NULL,
159  NULL, (imgID%2) ? &startWriteEvent2 :
160  &startWriteEvent1);
161  std::fill_n(tmpNodesIDptr, rowPitch*region[1], 0);
162  weCLQueue->enqueueUnmapMemObject((imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1,
163  tmpNodesIDptr);
164  #endif
165 
166 
167  // Note: if the current image is smaller than the previous one, part of the previous image
168  // is accessible since not overwritten by current image
170  region[0]=currImage.getWidth(); region[1]=currImage.getHeight();
171  region[2] = (nChannels<=4) ? 1 : nChannels;
172  std::copy(currImage.getData(), currImage.getData()+region[0]*region[1]*nChannels,
173  m_clTsImgPinnPtr+clPinnMemWOffset*region[0]*region[1]*nChannels);
174  if (nChannels<=4)
175  {
176  weCLQueue->enqueueWriteImage(*reinterpret_cast<cl::Image2D*>((imgID%2) ?
177  m_clTsImg2 : m_clTsImg1),
178  CL_FALSE,
179  origin, region, 0, 0,
180  (void*)(m_clTsImgPinnPtr +
181  clPinnMemWOffset*region[0]*region[1]*nChannels));
182  }
183  else
184  {
185  weCLQueue->enqueueWriteImage(*reinterpret_cast<cl::Image3D*>((imgID%2) ?
186  m_clTsImg2 : m_clTsImg1),
187  CL_FALSE,
188  origin, region, 0, 0,
189  (void*)(m_clTsImgPinnPtr +
190  clPinnMemWOffset*region[0]*region[1]*nChannels));
191  }
192 
193  region[2] = 1;
194  std::copy(currImage.getLabels(), currImage.getLabels()+region[0]*region[1],
195  m_clTsLabelsImgPinnPtr+clPinnMemWOffset*region[0]*region[1]);
196  weCLQueue->enqueueWriteImage((imgID%2) ? m_clTsLabelsImg2 : m_clTsLabelsImg1,
197  CL_FALSE,
198  origin, region, 0, 0,
199  (void*)(m_clTsLabelsImgPinnPtr +
200  clPinnMemWOffset*region[0]*region[1]));
201  std::copy(currImage.getSamples(), currImage.getSamples()+currImage.getNSamples(),
202  m_clTsSamplesBuffPinnPtr+clPinnMemWOffset*currImage.getNSamples());
203  weCLQueue->enqueueWriteBuffer((imgID%2) ? m_clTsSamplesBuff2 : m_clTsSamplesBuff1,
204  CL_FALSE,
205  0, currImage.getNSamples()*sizeof(cl_uint),
206  (void*)(m_clTsSamplesBuffPinnPtr +
207  clPinnMemWOffset*currImage.getNSamples()),
208  NULL, (imgID%2) ? &endWriteEvent2 : &endWriteEvent1);
209 
210  // Per-image prediction (i.e. compute samples end nodes)
211  if (currDepth!=1)
212  {
213  if (nChannels<=4)
214  {
215  m_clPredictKern.setArg(0, *reinterpret_cast<cl::Image2D*>((imgID%2) ?
216  m_clTsImg2 : m_clTsImg1));
217  }
218  else
219  {
220  m_clPredictKern.setArg(0, *reinterpret_cast<cl::Image3D*>((imgID%2) ?
221  m_clTsImg2 : m_clTsImg1));
222  }
223  m_clPredictKern.setArg(1, (imgID%2) ? m_clTsLabelsImg2 : m_clTsLabelsImg1);
224  m_clPredictKern.setArg(3, currImage.getWidth());
225  m_clPredictKern.setArg(4, currImage.getHeight());
226  m_clPredictKern.setArg(10, (imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1);
227  m_clPredictKern.setArg(11, (imgID%2) ? m_clPredictImg2 : m_clPredictImg1);
228 
229  for (unsigned int d=0; d<currDepth-1; d++)
230  {
231  weCLQueue->enqueueNDRangeKernel(m_clPredictKern,
232  cl::NullRange,
233  cl::NDRange(currImage.getWidth()+fillWidth,
234  currImage.getHeight()+fillHeight),
235  //cl::NDRange(WG_WIDTH, WG_HEIGHT),
236  cl::NDRange(WG_PREDICT_WIDTH, WG_PREDICT_HEIGHT),
237  NULL,
238  (d==0) ? ((imgID%2) ? &startComputeEvent2 :
239  &startComputeEvent1) : NULL);
240  weCLQueue->enqueueCopyImage((imgID%2) ? m_clPredictImg2 : m_clPredictImg1,
241  (imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1,
242  origin, origin, region);
243  }
244  }
245 
246 
247  // Per-image histogram computation
249  if (nChannels<=4)
250  {
251  m_clPerImgHistKern.setArg(0, *reinterpret_cast<cl::Image2D*>((imgID%2) ?
252  m_clTsImg2 : m_clTsImg1));
253  }
254  else
255  {
256  m_clPerImgHistKern.setArg(0, *reinterpret_cast<cl::Image3D*>((imgID%2) ?
257  m_clTsImg2 : m_clTsImg1));
258  }
259  m_clPerImgHistKern.setArg(2, currImage.getWidth());
260  m_clPerImgHistKern.setArg(3, currImage.getHeight());
261  m_clPerImgHistKern.setArg(4, (imgID%2) ? m_clTsLabelsImg2 : m_clTsLabelsImg1);
262  m_clPerImgHistKern.setArg(5, (imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1);
263  m_clPerImgHistKern.setArg(6, (imgID%2) ? m_clTsSamplesBuff2 : m_clTsSamplesBuff1);
264  m_clPerImgHistKern.setArg(7, currImage.getNSamples());
265  m_clPerImgHistKern.setArg(14, (imgID%2) ? m_clPerImgHistBuff2 : m_clPerImgHistBuff1);
266  m_clPerImgHistKern.setArg(16, startNode);
267  m_clPerImgHistKern.setArg(17, endNode);
268  weCLQueue->enqueueNDRangeKernel(m_clPerImgHistKern,
269  cl::NullRange,
270  cl::NDRange(currImage.getNSamples(), params.nFeatures),
271  //cl::NDRange(WG_WIDTH, WG_HEIGHT),
272  cl::NDRange(WG_LHIST_UPDATE_HEIGHT, WG_LHIST_UPDATE_WIDTH),
273  NULL, (imgID%2) ? &endComputeEvent2 : &endComputeEvent1);
274  }
275 
276  // ************ SECOND QUEUE: READ PREVIOUS RESULTS *************/
277  if (it!=tsImages.begin() && !m_skippedTsImg[imgID-1])
278  {
279  const TrainingSetImage<ImgType, nChannels> &prevImage = *(it-1);
280  fillWidth = (prevImage.getWidth()%WG_PREDICT_WIDTH) ?
281  WG_PREDICT_WIDTH-(prevImage.getWidth()%WG_PREDICT_WIDTH) : 0;
282  fillHeight = (prevImage.getHeight()%WG_PREDICT_HEIGHT) ?
283  WG_PREDICT_HEIGHT-(prevImage.getHeight()%WG_PREDICT_HEIGHT) : 0;
284  region[0]=prevImage.getWidth(); region[1]=prevImage.getHeight();
285 
286  // Producer
287  // Check if the queue is full
288  pthread_mutex_lock(&fifoMtx);
289  //if (fifoQueue.size()==GLOBAL_HISTOGRAM_FIFO_SIZE)
290  while (fifoQueue.size()==GLOBAL_HISTOGRAM_FIFO_SIZE)
291  {
292  //std::cout << "P: queue full, wait ..."<< std::endl;
293  pthread_cond_wait(&fifoCond, &fifoMtx);
294  }
295  pthread_mutex_unlock(&fifoMtx);
296 
297 
298  if (currDepth==1 && it==tsImages.begin())
299  {
300  std::fill_n(m_clTsNodesIDImgPinnPtr,
301  GLOBAL_HISTOGRAM_FIFO_SIZE*m_maxTsImgWidth*m_maxTsImgHeight, 0);
302  }
303  else
304  {
305  rCLQueue->enqueueReadImage((imgID%2) ? m_clTsNodesIDImg1 : m_clTsNodesIDImg2,
306  CL_TRUE,
307  origin, region, 0, 0,
308  (void*)(m_clTsNodesIDImgPinnPtr+queueIdx*m_maxTsImgWidth*m_maxTsImgHeight),
309  NULL, &startReadEvent);
310  }
311 
312  // Read per-image histogram
313  rCLQueue->enqueueReadBuffer((imgID%2) ? m_clPerImgHistBuff1 : m_clPerImgHistBuff2,
314  CL_TRUE,
315  0,
316  prevImage.getNSamples()*params.nFeatures*params.nThresholds*sizeof(cl_uchar),
317  (void*)(m_clPerImgHistBuffPinnPtr+queueIdx*perImgHistogramStride),
318  NULL, &endReadEvent);
319 
320  // Queue the current per-image histogram and predicted end nodes
321  pthread_mutex_lock(&fifoMtx);
322  fifoQueue.push(queueIdx);
323  //std::cout << "P: " << (queueIdx) << " produced" << std::endl;
324  queueIdx++;
325  queueIdx = queueIdx%GLOBAL_HISTOGRAM_FIFO_SIZE;
326  pthread_mutex_unlock(&fifoMtx);
327  pthread_cond_signal(&fifoCond);
328 
329  // Update timing info
330  cl_ulong startTime, endTime;
331 
332  startTime = (imgID%2) ? startWriteEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
333  startWriteEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>();
334  endTime = (imgID%2) ? endWriteEvent1.getProfilingInfo<CL_PROFILING_COMMAND_END>() :
335  endWriteEvent2.getProfilingInfo<CL_PROFILING_COMMAND_END>();
336  totWriteTime += endTime-startTime;
337 
338  startTime = (currDepth!=1) ?
339  ((imgID%2) ? startComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
340  startComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>()):
341  ((imgID%2) ? endComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
342  endComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>());
343  endTime = (imgID%2) ? endComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_END>() :
344  endComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_END>();
345  totComputeTime += endTime-startTime;
346 
347  startTime = (currDepth!=1) ?
348  startReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
349  endReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_START>();
350  endTime = endReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_END>();
351  totReadTime += endTime-startTime;
352  }
353 
354  // Sync point: wait for computation to finish (reading is finished for sure since
355  // local histogram read is blocking)
356  if (!m_skippedTsImg[imgID])
357  if (imgID%2) endComputeEvent2.wait(); else endComputeEvent1.wait();
358  }
359 
360  //************ Last image **********
361  --imgID;
362  if (!m_skippedTsImg[imgID])
363  {
364  const TrainingSetImage<ImgType, nChannels> &currImage = *(tsImages.end()-1);
365  cl::CommandQueue *rCLQueue = (imgID%2) ? &m_clQueue2 : &m_clQueue1;
366 
367  fillWidth = (currImage.getWidth()%WG_PREDICT_WIDTH) ?
368  WG_PREDICT_WIDTH-(currImage.getWidth()%WG_PREDICT_WIDTH) : 0;
369  fillHeight = (currImage.getHeight()%WG_PREDICT_HEIGHT) ?
370  WG_PREDICT_HEIGHT-(currImage.getHeight()%WG_PREDICT_HEIGHT) : 0;
371  region[0]=currImage.getWidth(); region[1]=currImage.getHeight();
372 
373 
374  pthread_mutex_lock(&fifoMtx);
375  while (fifoQueue.size()==GLOBAL_HISTOGRAM_FIFO_SIZE)
376  {
377  //std::cout << "P: queue full, wait ..."<< std::endl;
378  pthread_cond_wait(&fifoCond, &fifoMtx);
379  }
380  pthread_mutex_unlock(&fifoMtx);
381 
382  if (currDepth!=1)
383  {
384  rCLQueue->enqueueReadImage((imgID%2) ? m_clTsNodesIDImg2 : m_clTsNodesIDImg1,
385  CL_FALSE,
386  origin, region, 0, 0,
387  (void*)(m_clTsNodesIDImgPinnPtr+queueIdx*m_maxTsImgWidth*m_maxTsImgHeight),
388  NULL, &startReadEvent);
389  }
390 
391  // Read per-image histogram
392  rCLQueue->enqueueReadBuffer((imgID%2) ? m_clPerImgHistBuff2 : m_clPerImgHistBuff1,
393  CL_TRUE,
394  0,
395  currImage.getNSamples()*params.nFeatures*params.nThresholds*sizeof(cl_uchar),
396  (void*)(m_clPerImgHistBuffPinnPtr+queueIdx*perImgHistogramStride),
397  NULL, &endReadEvent);
398 
399  // Queue the current per-image histogram and predicted end nodes
400  pthread_mutex_lock(&fifoMtx);
401  fifoQueue.push(queueIdx);
402  //std::cout << "P: " << (queueIdx) << " produced" << std::endl;
403  queueIdx++;
404  queueIdx = queueIdx%GLOBAL_HISTOGRAM_FIFO_SIZE;
405  pthread_mutex_unlock(&fifoMtx);
406  pthread_cond_signal(&fifoCond);
407 
408  cl_ulong startTime, endTime;
409 
410  startTime = (imgID%2) ? startWriteEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
411  startWriteEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>();
412  endTime = (imgID%2) ? endWriteEvent2.getProfilingInfo<CL_PROFILING_COMMAND_END>() :
413  endWriteEvent1.getProfilingInfo<CL_PROFILING_COMMAND_END>();
414  totWriteTime += endTime-startTime;
415 
416  startTime = (currDepth!=1) ?
417  ((imgID%2) ? startComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
418  startComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>()):
419  ((imgID%2) ? endComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
420  endComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_START>());
421  endTime = (imgID%2) ? endComputeEvent2.getProfilingInfo<CL_PROFILING_COMMAND_END>() :
422  endComputeEvent1.getProfilingInfo<CL_PROFILING_COMMAND_END>();
423  totComputeTime += endTime-startTime;
424 
425  startTime = (currDepth!=1) ?
426  startReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_START>() :
427  endReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_START>();
428  endTime = endReadEvent.getProfilingInfo<CL_PROFILING_COMMAND_END>();
429  totReadTime += endTime-startTime;
430  }
431 
432 
433  // DONE with local histograms
434  pthread_join(consumer, NULL);
435 
436 
437  double totTime = static_cast<double>(totWriteTime)*1.e-9;
438  BOOST_LOG_TRIVIAL(info) << "Total local histogram write time: "
439  << totTime
440  << " seconds";
441  // << (avg: " << totTime/tsImages.size()
442  // << " seconds)";
443  totTime = static_cast<double>(totComputeTime)*1.e-9;
444  BOOST_LOG_TRIVIAL(info) << "Total local histogram compute time: "
445  << totTime
446  << " seconds";
447  // << (avg: " << totTime/tsImages.size()
448  // << " seconds)";
449  totTime = static_cast<double>(totReadTime)*1.e-9;

460 template <typename ImgType, unsigned int nChannels, typename FeatType, unsigned int FeatDim,
461  unsigned int nClasses>
462 void *_updateGlobalHistogram(void *_data)
463 {
464  ConsumerProducerData<ImgType, nChannels, FeatType, FeatDim, nClasses> *data =
465  (ConsumerProducerData<ImgType, nChannels, FeatType, FeatDim, nClasses>*)_data;
466 
467  int *nodesIDImg = data->nodesIDImg;
468  unsigned int maxImgWidth = data->maxImgWidth;
469  unsigned int maxImgHeight = data->maxImgHeight;
470  unsigned int *perNodeTotSamples = data->perNodeTotSamples;
471  unsigned int *perClassTotSamples = data->perClassTotSamples;
472  boost::unordered_map<int, int> *frontierIdxMap = data->frontierIdxMap;
473  unsigned int **histogram = data->histogram;
474  unsigned char *perImgHistogram = data->perImgHistogram;
475  size_t perImgHistogramStride = data->perImgHistogramStride;
476  const TrainingSet<ImgType, nChannels> &trainingSet = *data->trainingSet;
477  bool *skippedTsImg = data->skippedTsImg;
478  bool *toSkipTsImg = data->toSkipTsImg;
479  const TreeTrainerParameters<FeatType, FeatDim> &params = *data->params;
480  unsigned int currDepth = data->currDepth;
