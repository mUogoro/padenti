53 template <typename type, unsigned int nChannels>
54 unsigned int ClassWeightedImageSampler<type, nChannels>::sample(const type *data, const unsigned char *labels,
55  unsigned int width, unsigned int height,
56  unsigned int *samples) const
57 {
58  unsigned int **perClassBucket = new unsigned int*[m_nClasses];
59  for (int i=0; i<m_nClasses; i++) perClassBucket[i] = new unsigned int[m_perClassBucketSize[i]];
60 
61  unsigned int *perClassNSamples = new unsigned int[m_nClasses];
62  unsigned int *perClassCounter = new unsigned int[m_nClasses];
63  std::fill(perClassNSamples, perClassNSamples+m_nClasses, 0);
64  std::fill(perClassCounter, perClassCounter+m_nClasses, 0);
65 
66  boost::random::mt19937 gen;
67  boost::random::uniform_01<> dist;
68 
69  boost::crc_32_type checksum_agent;
70  checksum_agent.process_bytes((unsigned char*)data, sizeof(type)*width*height*nChannels);
71  gen.seed(checksum_agent.checksum() ^ this->m_seed);
72 
73 
74  // Scan the whole image and perform per-class reservoir sampling
75  for (int i=0; i<width*height; i++)
76  {
77  unsigned char label = labels[i];
78  if (label)
79  {
80  label -= 1;
81  unsigned int *currBucket = perClassBucket[label];
82  unsigned int perClassBucketSize = m_perClassBucketSize[label];
83 
84  if (perClassNSamples[label]<perClassBucketSize)
85  {
86  currBucket[perClassNSamples[label]] = i;
87  perClassNSamples[label]++;
88  }
89  else
90  {
91  unsigned int j = roundf(dist(gen)*perClassCounter[label]);
92  if (j<perClassBucketSize)
93  {
94  currBucket[j] = i;
95  }
96  }
97  perClassCounter[label]++;
98  }
99  }
100 
101  // Copy the sampled pixels to the output buffer
102  unsigned int totSamples=perClassNSamples[0];
103  std::copy(perClassBucket[0], perClassBucket[0]+perClassNSamples[0], samples);
104  for (int i=1; i<m_nClasses; i++)
105  {
106  std::copy(perClassBucket[i], perClassBucket[i]+perClassNSamples[i],
107  samples+totSamples);
108  totSamples += perClassNSamples[i];
109  }
110 
111  // Finally, sort samples
112  std::sort(samples, samples+totSamples);
113 
114  for (int i=0; i<m_nClasses; i++) delete perClassBucket[i];
115  delete []perClassBucket;
116  delete []perClassNSamples;
117  delete []perClassCounter;
118 
119  return totSamples;
120 }
